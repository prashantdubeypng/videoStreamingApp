UserName availability check --> indexing the Database --> username ->index
users {
    id (PK, UUID or BIGINT)

    username (UNIQUE, INDEX, NOT NULL)
    email (UNIQUE, INDEX, NULLABLE)
    mobile (UNIQUE, INDEX, NULLABLE)

    name

    password_hash (NOT NULL)

    is_active (BOOLEAN DEFAULT true)
    is_verified (BOOLEAN DEFAULT false)

    created_at
    updated_at
}
refresh_tokens {
    id (PK)

    user_id (FK → users.id, INDEX)

    token_hash (NOT raw token)

    expiry_at

    is_revoked (BOOLEAN DEFAULT false)

    created_at
    updated_at
}
